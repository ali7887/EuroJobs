import { NextRequest, NextResponse } from "next/server";
import { jobService } from "@/lib/services/job.service";
// import { withAuth } from "@/app/api/with-auth";
// import { requireRole } from "@/lib/auth/require-role";
// import { AuthContext } from "@/lib/auth/auth.context";

export async function GET(
  req: NextRequest,
  { params }: { params: { jobId: string } }
) {
  const job = await jobService.getById(Number(params.jobId));

  if (!job) {
    return NextResponse.json(
      { message: "Job not found" },
      { status: 404 }
    );
  }

  return NextResponse.json({ data: job });
}

export async function DELETE(
  req: NextRequest,
  { params }: { params: { jobId: string } }
) {

  return NextResponse.json({
    success: true,
  });
}

/* ===== FILE: src/app/api/jobs/[jobId]/apply/route.ts ===== */

import { NextRequest, NextResponse } from "next/server";
import { withAuth } from "@/app/api/with-auth";
import { requireRole } from "@/lib/auth/require-role";
import { applicationService } from "@/lib/services/application.service";
import { AuthContext } from "@/lib/auth/auth.context";
type AuthUser = {
  userId: number;
  email: string;
  role: string;
};

export const POST = withAuth(
  async (
    req: NextRequest,
    user: AuthContext,   // ← FIX
    { params }: { params: { jobId: string } }
  ) => {

    requireRole(user, ["user"]);

    const body = await req.json();

    const application = await applicationService.applyToJob(
      Number(user.userId),
      Number(params.jobId),
      body
    );

    return NextResponse.json({
      success: true,
      data: application
    });
  }
);

/* ===== FILE: src/app/api/jobs/[jobId]/applications/route.ts ===== */

import { NextRequest, NextResponse } from "next/server";
import { applicationService } from "@/lib/services/application.service";

type RouteParams = {
  params: {
    jobId: string;
  };
};

export const runtime = "nodejs";

export async function GET(
  _req: NextRequest,
  { params }: RouteParams,
) {
  const jobIdNum = Number(params.jobId);

  if (Number.isNaN(jobIdNum)) {
    return NextResponse.json({ error: "Invalid jobId" }, { status: 400 });
  }

  const applications = await applicationService.getJobApplications(jobIdNum);

  return NextResponse.json(applications);
}
